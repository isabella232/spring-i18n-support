5   package com.namics.oss.spring.support.i18n.config;
6   
7   import com.namics.oss.spring.support.i18n.DaoMessageSource;
8   import com.namics.oss.spring.support.i18n.dao.MessageSourceDao;
9   import com.namics.oss.spring.support.i18n.dao.MessageSourceManagementDao;
10  import com.namics.oss.spring.support.i18n.excel.ExcelReader;
11  import com.namics.oss.spring.support.i18n.excel.ExcelWriter;
12  import com.namics.oss.spring.support.i18n.service.MessageManagementService;
13  import com.namics.oss.spring.support.i18n.service.MessageManagementServiceImpl;
14  import com.namics.oss.spring.support.i18n.text.SqlScriptWriter;
15  import org.springframework.context.annotation.Bean;
16  import org.springframework.jmx.export.annotation.AnnotationMBeanExporter;
17  
18  /**
19   * I18nConfigTemplate.
20   *
21   * @author aschaefer, Namics AG
22   * @since 18.03.14 16:58
23   */
24  public class I18nConfigTemplate
25  {
26  	@Bean
27  	public AnnotationMBeanExporter annotationMBeanExporter() {
28  		return new AnnotationMBeanExporter();
29  	}
30  
31  	@Bean
32  	public ExcelReader excelReader()
33  	{
34  		return new ExcelReader();
35  	}
36  
37  	@Bean
38  	public ExcelWriter excelWriter()
39  	{
40  		return new ExcelWriter();
41  	}
42  
43  	@Bean
44  	public SqlScriptWriter sqlScriptWriter()
45  	{
46  		return new SqlScriptWriter();
47  	}
48  
49  	@Bean
50  	public DaoMessageSource i18nMessageSource(MessageSourceDao messageSourceDao)
51  	{
52  		DaoMessageSource source = new DaoMessageSource();
53  		source.setMessageSourceDao(messageSourceDao);
54  		return source;
55  	}
56  
57  	@Bean
58  	public MessageManagementService messageManagementService(MessageSourceManagementDao messageSourceDao)
59  	{
60  		MessageManagementServiceImpl service = new MessageManagementServiceImpl();
61  		service.setExcelReader(excelReader());
62  		service.setExcelWriter(excelWriter());
63  		service.setMessageSourceDao(messageSourceDao);
64  		return service;
65  	}
66  }
