5   package com.namics.oss.spring.support.i18n.demo.config;
6   
7   import org.springframework.context.annotation.Bean;
8   import org.springframework.context.annotation.Configuration;
9   import org.springframework.jdbc.datasource.embedded.EmbeddedDatabase;
10  import org.springframework.jdbc.datasource.embedded.EmbeddedDatabaseBuilder;
11  import org.springframework.jdbc.datasource.embedded.EmbeddedDatabaseType;
12  
13  import javax.sql.DataSource;
14  
15  /**
16   * WebappConfig.
17   *
18   * @author aschaefer, Namics AG
19   * @since 18.03.14 17:15
20   */
21  @Configuration
22  public class WebappConfig {
23  	@Bean
24  	public DataSource dataSource() {
25  		EmbeddedDatabaseBuilder builder = new EmbeddedDatabaseBuilder();
26  		EmbeddedDatabase db = builder.setType(EmbeddedDatabaseType.H2)
27  		                             .addScript("/jdbc/schema.sql")
28  		                             .addScript("/jdbc/data.sql")
29  		                             .build();
30  		return db;
31  	}
32  }
